<script>
const area=document.getElementById('area');
const countEl=document.getElementById('count');
const end=document.getElementById('end');
let count=0;const target=5;let interval;
function random(min,max){return Math.floor(Math.random()*(max-min+1))+min}
function spawnHeart(){
  const h=document.createElement('div');h.className='heart';h.innerHTML='❤️';
  h.style.left=random(20,area.clientWidth-20)+'px';
  h.style.top=random(20,area.clientHeight-20)+'px';
  area.appendChild(h);
  h.onclick=()=>{h.remove();count++;countEl.textContent=count;if(count>=target)showEnd()}
  setTimeout(()=>h.remove(),3000);
}
function startGame(){
  count=0;countEl.textContent=0;end.classList.add('hidden');area.innerHTML='';
  interval=setInterval(spawnHeart,700)
}
function showEnd(){clearInterval(interval);area.innerHTML='';end.classList.remove('hidden')}
startGame();
</script>
